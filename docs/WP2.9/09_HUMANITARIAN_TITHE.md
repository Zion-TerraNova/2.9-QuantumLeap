### Smart Contract Logic

```python
# Simplified tithe calculation (actual in src/core/)
def calculate_tithe(block_reward: float, current_year: int) -> dict:
    """Calculate humanitarian tithe based on escalation schedule."""
    
    tithe_rates = {
        (1, 5): 0.10,
        (6, 10): 0.12,
        (11, 15): 0.15,
        (16, 25): 0.18,
        (26, 35): 0.22,
        (36, 45): 0.25
    }
    
    rate = 0.10  # Default
    for (start, end), r in tithe_rates.items():
        if start <= current_year <= end:
            rate = r
            break
    
    tithe_amount = block_reward * rate
    
    return {
        "total_tithe": tithe_amount,
        "humanita": tithe_amount * 0.60,  # 60% for humans
        "hanuman": tithe_amount * 0.40,   # 40% for animals
    }
```
